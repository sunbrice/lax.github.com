计算两个整数最大公约数和最小公倍数的算法：C++语言实现

#include <iostream>
using namespace std;
int gcd(int a, int b) //最大公约数
{
while(a!=b)
{ a>b ? a -= a/b * b : b -= b/a * a; }
return a;
}
int lcm(int a, int b) //最小公倍数
{
return a * b / gcd(a,b);
}
int main()
{
int x, y;
cout << "请输入两个正整数：" <<endl;
cin >> x >> y;
cout << "最大公约数： " << gcd(x,y) << " ； 最小公倍数： " << lcm(x,y) << endl;
return 0;
}
